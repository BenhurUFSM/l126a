1. Faça uma função que recebe o tamanho de um vetor, um vetor de inteiros desse tamanho e um inteiro. A função deve retornar um `bool` que diz se o inteiro está no vetor.
1. Como o anterior, mas a função retorna o número de vezes que o inteiro aparece no vetor.
1. Como o anterior, mas a função retorna a primeira posição onde o inteiro aparece no vetor, ou -1 caso não exista o inteiro no vetor.
1. Como o anterior, mas a função retorna a última posição.
1. Repita os 4 exercícios anteriores, com a diferença que cada elemento do vetor tem um registro com 2 inteiros, representando um intervalo de valores. Um inteiro é considerado como pertencente a um intervalo se for maior ou igual ao primeiro inteiro do intervalo e menor ou igual ao segundo. Por exemplo, se a função está buscando pelo número 5, um registro que contenha <5,5> ou um que contenha <5,8> ou um que contenha <3,7> servem para dizer que o 5 está no vetor.
1. Repita os 4 exercícios novamente, mas agora cada elemento tem 2 intervalos, e a busca é por um registro que tem 2 valores. O primeiro valor deve estar no primeiro intervalo e o segundo valor deve estar no segundo intervalo. Por exemplo, se a busca é por <3,7>, o valor <1,5><4,10> serve, mas <1,5><4,6> não. Use os tipos abaixo.
   ```c
   typedef struct {
       int inferior;
       int superior;
   } intervalo;
   typedef struct {
       intervalo i1;
       intervalo i2;
   } intervalo_duplo;
   typedef struct {
   ```

1. Faça uma função que recebe o tamanho de um vetor e um vetor de "postits", e um ponto. Os postits estão em ordem em que foram colocados no espaço, os últimos tendo sido colocados sobre os primeiros, caso compartilhem posição. O função deve retornar o índice do vetor onde está o último postit colocado, que contenha o ponto. Caso não tenha nenhum postit na posição do ponto, a função deve retornar -1. (visto em aula, resultou mais ou menos no programa abaixo)
   ```c
#include <stdio.h>
#include <stdbool.h>

typedef struct {
  int r;  // nível de vermelho, 0-255
  int g;  // verde
  int b;  // azul
} cor;

typedef struct {
  int x;  // distância da esquerda
  int y;  // distância de baixo
} ponto;

typedef struct {
  ponto inf_esq;  // ponto inferior esquerdo
  ponto sup_dir;  // ponto superior direito
} retangulo;

typedef struct {
  retangulo r;
  cor c;
  char t[20];
} postit;

bool ponto_no_retangulo(ponto p, retangulo r)
{
  if (p.x < r.inf_esq.x) return false;
  if (p.x > r.sup_dir.x) return false;
  if (p.y < r.inf_esq.y) return false;
  if (p.y > r.sup_dir.y) return false;
  return true;
}

// retorna índice do último postit em v que contém p
// ou -1 se não houver postit nesse ponto.
int acha_postit(int n, postit v[n], ponto p)
{
  for (int i = n-1; i >= 0; i-=1) {
    postit post = v[i];
    retangulo ret = post.r;
    if (ponto_no_retangulo(p, ret)) {
      return i;
    }
  }
  return -1;
}

void cor_letra(cor c)
{
  printf("%c[38;2;%d;%d;%dm", 27, c.r, c.g, c.b);
}

void cor_fundo(cor c)
{
  printf("%c[48;2;%d;%d;%dm", 27, c.r, c.g, c.b);
}

void cor_normal()
{
  printf("%c[0m", 27);
}

void desenha_postits(int n, postit v[n])
{
  char letra;
  for (int lin = 23; lin >= 0; lin-=1) {
    for (int col = 0; col < 80; col++) {
      ponto pos = { col, lin };
      int i = acha_postit(n, v, pos);
      if (i != -1) {
        cor_fundo(v[i].c);
        cor_letra(v[i].c);
        letra = v[i].t[0];
      } else {
        cor_fundo((cor){0,0,0});
        letra = ' ';
      }
      putchar(letra);
    }
    cor_normal();
    putchar('\n');
  }
}

int main()
{
  postit v[] = {
    {{{ 0, 0}, {49,23}}, {  0,  0, 50}, "escuro"  },
    {{{15, 9}, {40,30}}, { 20, 20,200}, "azul"    },
    {{{10,10}, {20,20}}, {200,  0,  0}, "vermelho"},
    {{{40,15}, {75,22}}, {200,140,  0}, "laranja" },
  };
  desenha_postits(4, v);
}
   ```
